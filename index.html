    <script>

        document.addEventListener('DOMContentLoaded', () => {

            // --- DOM Elements ---

            const receiverNameInput = document.getElementById('receiverName');

            const receiverImageInput = document.getElementById('receiverImage');

            const messageTextInput = document.getElementById('messageText');

            const addMessageBtn = document.getElementById('addMessageBtn');

            const downloadBtn = document.getElementById('downloadBtn');

            const resetBtn = document.getElementById('resetBtn');

            const themeSelector = document.getElementById('theme-selector');

            const chatContainer = document.getElementById('chat-container');

            const headerName = document.getElementById('header-name');

            const headerImg = document.getElementById('header-img');

            const chatMessages = document.getElementById('chat-messages');

            const receiverRadioLabel = document.getElementById('receiverRadioLabel');

            const lastMessageMeta = () => chatMessages.querySelector('.sender-bubble:last-child .bubble-meta');


            // --- Functions ---

            const updateUI = () => {

                const name = receiverNameInput.value.trim();

                headerName.textContent = name || 'Receiver';

                receiverRadioLabel.textContent = name || 'Receiver';

                if (!receiverImageInput.files[0]) {

                    headerImg.src = `https://placehold.co/40x40/E2E8F0/4A5568?text=${name.charAt(0) || 'R'}`;

                }

            };


            const handleImageUpload = (event) => {

                const file = event.target.files[0];

                if (file) {

                    const reader = new FileReader();

                    reader.onload = e => headerImg.src = e.target.result;

                    reader.readAsDataURL(file);

                }

            };

            

            const handleThemeChange = (event) => {

                const theme = event.target.value;

                chatContainer.className = chatContainer.className.replace(/whatsapp-theme|instagram-theme/g, '');

                chatContainer.classList.add(`${theme}-theme`);

                updateSeenStatus();

            };


            const updateSeenStatus = () => {

                const lastSenderBubble = lastMessageMeta();

                if (lastSenderBubble) lastSenderBubble.style.display = 'flex';

            };


            const addMessage = (event) => {

                event.preventDefault(); // Prevent default button action

                const text = messageTextInput.value.trim().replace(/\n/g, '<br>');

                if (!text) return;


                const senderType = document.querySelector('input[name="sender"]:checked').value;

                const messageElement = document.createElement('div');

                const time = new Date().toLocaleTimeString('en-US', { hour: 'numeric', minute: 'numeric', hour12: true });


                const lastMeta = lastMessageMeta();

                if (lastMeta) lastMeta.style.display = 'none';


                if (senderType === 'sender') {

                    messageElement.className = 'flex justify-end';

                    messageElement.innerHTML = `

                        <div class="bubble sender-bubble">

                            <p>${text}</p>

                            <div class="bubble-meta flex items-center justify-end">

                                <span>${time}</span>

                                <span class="blue-ticks ml-1">✓✓</span>

                            </div>

                        </div>`;

                } else {

                    messageElement.className = 'flex justify-start';

                    messageElement.innerHTML = `

                        <div class="bubble receiver-bubble">

                            <p>${text}</p>

                            <div class="bubble-meta flex items-center justify-end" style="display:none">

                                <span>${time}</span>

                            </div>

                        </div>`;

                }


                chatMessages.appendChild(messageElement);

                messageTextInput.value = '';

                chatMessages.parentElement.scrollTop = chatMessages.parentElement.scrollHeight;

                updateSeenStatus();

            };


            const downloadChatAsImage = (event) => {

                event.preventDefault(); // Prevent default button action

                const originalShadow = chatContainer.style.boxShadow;

                chatContainer.style.boxShadow = 'none';

                

                html2canvas(chatContainer, { 

                    useCORS: true, 

                    backgroundColor: null,

                    logging: false,

                    scale: 2

                }).then(canvas => {

                    const link = document.createElement('a');

                    link.download = 'chat-conversation.png';

                    link.href = canvas.toDataURL('image/png');

                    link.click();

                    chatContainer.style.boxShadow = originalShadow;

                }).catch(err => {

                    console.error("Image capture failed:", err);

                    alert("Sorry, could not download the image.");

                    chatContainer.style.boxShadow = originalShadow;

                });

            };


            const resetChat = (event) => {

                event.preventDefault(); // Prevent default button action

                if (confirm("Are you sure you want to reset everything?")) {

                    receiverNameInput.value = 'Jessica';

                    receiverImageInput.value = '';

                    document.querySelector('input[name="theme"][value="whatsapp"]').checked = true;

                    handleThemeChange({ target: { value: 'whatsapp' } });

                    chatMessages.innerHTML = '';

                    updateUI();

                }

            };


            // --- Event Listeners ---

            receiverNameInput.addEventListener('input', updateUI);

            receiverImageInput.addEventListener('change', handleImageUpload);

            addMessageBtn.addEventListener('click', addMessage);

            downloadBtn.addEventListener('click', downloadChatAsImage);

            resetBtn.addEventListener('click', resetChat);

            themeSelector.addEventListener('change', handleThemeChange);

            

            // Initial call

            updateUI();

        });

    </script>
